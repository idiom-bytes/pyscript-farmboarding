V4 Marketplace TVL #pyodide does not support requests #https://blog.devgenius.io/pyscript-making-http-requests-best-approach-tutorial-8ae062438335 from pyodide.http import pyfetch import asyncio CHAIN_IDS = [1] _CHAINID_TO_NETWORK = { 8996: "development", # ganache 1: "mainnet", 3: "ropsten", 4: "rinkeby", 56: "bsc", 137: "polygon", 246: "energyweb", 1287: "moonbase", 1285: "moonriver", 80001: "mumbai", } def chainIdToNetwork(chainID): """Returns the network name for a given chainID""" return _CHAINID_TO_NETWORK[chainID] def chainIdToSubgraphUri(chainID): """Returns the subgraph URI for a given chainID""" sg = "/subgraphs/name/oceanprotocol/ocean-subgraph" network_str = chainIdToNetwork(chainID) return f"https://v4.subgraph.{network_str}.oceanprotocol.com" + sg async def submitQuery(query, chainID): """Submits GQL query to subgraph""" subgraph_url = chainIdToSubgraphUri(chainID) try: response = await pyfetch(url=subgraph_url, method="POST", json={"query": query}) if response.ok: return response.json() except JsException: return None async def main(): for chainID in CHAIN_IDS : query = """ { orders(first:1000) { id, datatoken { id } lastPriceToken, lastPriceValue estimatedUSDValue, block } } """ result = await submitQuery(query, chainID) print(f"Number of orders onchainID {chainID} is: {len(result)}") loop = asyncio.get_event_loop() loop.run_until_complete( main() )
