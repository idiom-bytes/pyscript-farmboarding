HTTP requests via pyfetch from the v4.subgraph

from pyodide.http import pyfetch import asyncio import json subgraph_url = "https://v4.subgraph.polygon.oceanprotocol.com/subgraphs/name/oceanprotocol/ocean-subgraph" query = """ { orders(first:1000) { id, datatoken { id } lastPriceToken, lastPriceValue estimatedUSDValue, block } } """ response = await pyfetch( url=subgraph_url, method='POST', body=json.dumps({"query": query}) ) output = f"GET request=> status:{response.status}, json:{await response.json()}" print(f"{output}")
